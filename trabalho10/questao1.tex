\subsection*{1. Dê uma prova formal de $A \rightarrow B$, em que $A$ e $B$ são definidos como segue:}

\[
\begin{aligned}
A &= \exists x \big(r(x) \land \forall y \big(p(y) \rightarrow q(x, y)\big)\big) 
   \;\land\;
   \forall x \big(r(x) \rightarrow \forall y \big(s(y) \rightarrow \lnot q(x,y)\big)\big) \\[8pt]
B &= \forall x \big(p(x) \rightarrow \lnot s(x)\big)
\end{aligned}
\]

\begin{flalign*}
	 & 1. \exists x (r(x) \land \forall y (p(y) \rightarrow q(x, y))) &  & \text{P}\\
	 & 2. \forall x (r(x) \rightarrow \forall y (s(y) \rightarrow \lnot q(x,y))) &  & \text{P}\\
	 & 3. r(c) \land \forall y (p(y) \rightarrow q(c, y)) &  & \text{1, IE}\\
     & 4. \forall y (p(y) \rightarrow q(c,y)) & & \text{3, Simp.} \\
     & 5. p(y) \rightarrow q(c,y) & & \text{4, IU} \\
     & 6. r(c) \rightarrow \forall y (s(y) \rightarrow \lnot q(c,y)) & & \text{2, IU} \\
     & 7. \quad \quad r(c) & & \text{P (Para $s(y) \rightarrow \lnot q(c,y)$)} \\
     & 8. \quad \quad \forall y (s(y) \rightarrow \lnot q(c,y)) & & \text{6, 7 MP} \\
     & 9. s(y) \rightarrow \lnot q(c,y) & & \text{8, IU} \\
     & 10. \quad \quad  p(y) & & \text{P (Para $q(c, y)$)} \\
     & 11. q(c, y) & & \text{5, 10, MP} \\
     & 12. \quad \quad  s(y) & & \text {P (Para obter contradição)} \\
     & 13. \quad \quad  \lnot q(c, y) & & \text {9, 12, MP} \\
     & 14. \quad \quad  Falso & & \text {11, 13, Contr.} \\
     & 15. \lnot s(y) & & {12 - 14, PI} \\
     & 16. p(y) \rightarrow \lnot s(y) & & {10-15, PC}\\
     & 17. \forall y (p(y) \rightarrow \lnot s(y)) & & {10-15, PC}\\
     & 18. \forall x (p(x) \rightarrow \lnot s(x)) & & {17, Subs.}\\
     QED
\end{flalign*}
